API DOC

REGISTER
POST /auth/register
BODY firstName, lastName, emailOrMobile, passsword, confirmPassword
RESPONSE 
201 
{ 
    accessToken : String, 
    user : { 
        id : Number, 
        firstName : String, 
        lastName : String, 
        email : String, 
        mobile : String, 
        password : String,
        profileImage : String, 
        coverImage : String 
    }
}
400 { message : String, emailOrMobileExist: true }
500 { message : String }
// In todolist-lab-backend, P Earth only sends accessToken via only feature login
// In cc15-fakebuck-mvc-express, P Earth sends accessToken via both register and login features, so you will be logged in automatically when you are done registering

LOGIN
POST /auth/login
BODY emailOrMobile, password
RESPONSE 
200 
{ 
    accessToken : String, 
    user : { 
        id : Number, 
        firstName : String, 
        lastName : String, 
        email : String, 
        mobile : String, 
        password : String,
        profileImage : String, 
        coverImage : String 
    }
}
400 { message : String }
500 { message : String }

GET AUTHENTICATED USER (verify token)
GET /auth/me
HEADERS AUTHORIZATION Bearer ACCESS_TOKEN
RESPONSE
200 { user: { id: Number, firstName: String, ... } }
401 { message: String }
500 { message: String }

UPDATE PROFILE/COVER IMAGE
PATCH /user
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
BODY profileImage?, coverImage?
RESPONSE
200 { profileImage?: String, coverImage?: String }
400 { message: String }
401 { message: String } // TokenExpired, Invalid token
500 { message: String }

GET USER PROFILE BY id
GET /user/:userId
HEADER AUTHORIZATION Bearer ACCESS_TOKEN
RESPONSE
200 { 
  user: { id: Number, firstName: String, ... }, 
  status: String, 
  friends: [ {id: Number, firstName: String, ...} ] 
}
401 { message: String }
500 { message: String }
